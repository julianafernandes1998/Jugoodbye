      const segundos = Math.floor((diferenca / 1000) % 60);

      document.getElementById("hours").innerHTML = formatar(horas);

      document.getElementById("minutes").innerHTML = formatar(minutos);

      document.getElementById("seconds").innerHTML = formatar(segundos);

    }
